204 @staticmethod 

205 def _remove_duplicate_entities( 

206 results: List[dict], dup_pix_tolerance: int = 5 

207 ) -> List[dict]: 

208 """Handle when a word is detected multiple times as different types of entities. 

209 

210 :param results: List of detected PHI with bbox info. 

211 :param dup_pix_tolerance: Pixel difference tolerance for identifying duplicates. 

212 :return: Detected PHI with no duplicate entities. 

213 """ 

214 dups = [] 

215 sorted(results, key=lambda x: x["score"], reverse=True) 

216 results_no_dups = [] 

217 dims = ["left", "top", "width", "height"] 

218 

219 # Check for duplicates 

220 for i in range(len(results) - 1): 

221 i_dims = {dim: results[i][dim] for dim in dims} 

222 

223 # Ignore if we've already detected this dup combination 

224 for other in range(i + 1, len(results)): 

225 if i not in results_no_dups: 

226 other_dims = {dim: results[other][dim] for dim in dims} 

227 matching_dims = { 

228 dim: abs(i_dims[dim] - other_dims[dim]) <= dup_pix_tolerance 

229 for dim in dims 

230 } 

231 matching = list(matching_dims.values()) 

232 

233 if all(matching): 

234 lower_scored_index = ( 

235 other 

236 if results[other]["score"] < results[i]["score"] 

237 else i 

238 ) 

239 dups.append(lower_scored_index) 

240 

241 # Remove duplicates 

242 for i in range(len(results)): 

243 if i not in dups: 

244 results_no_dups.append(results[i]) 

245 

246 return results_no_dups

248 def _label_all_positives( 

249 self, 

250 gt_labels_dict: dict, 

251 ocr_results: List[dict], 

252 detected_phi: List[dict], 

253 tolerance: int = 50, 

254 ) -> List[dict]: 

255 """Label all entities detected as PHI by using ground truth and OCR results. 

256 

257 All positives (detected_phi) do not contain PHI labels and are thus 

258 difficult to work with intuitively. This method maps back to the 

259 actual PHI to each detected sensitive entity. 

260 

261 :param gt_labels_dict: Dictionary with ground truth labels for a 

262 single DICOM instance. 

263 :param ocr_results: All detected text. 

264 :param detected_phi: Formatted analyzer_results. 

265 :param tolerance: Tolerance for exact coordinates and size data. 

266 :return: List of all positives, labeled. 

267 """ 

268 all_pos = [] 

269 

270 # Cycle through each positive (TP or FP) 

271 for analyzer_result in detected_phi: 

272 # See if there are any ground truth matches 

273 all_pos, gt_match_found = self.bbox_processor.match_with_source( 

274 all_pos, gt_labels_dict, analyzer_result, tolerance 

275 ) 

276 

277 # If not, check back with OCR 

278 if not gt_match_found: 

279 all_pos, _ = self.bbox_processor.match_with_source( 

280 all_pos, ocr_results, analyzer_result, tolerance 

281 ) 

282 

283 # Remove any duplicates 

284 all_pos = self._remove_duplicate_entities(all_pos) 

285 

286 return all_pos

289 def calculate_precision(gt: List[dict], all_pos: List[dict]) -> float: 

290 """Calculate precision. 

291 

292 :param gt: List of ground truth labels. 

293 :param all_pos: All Detected PHI (mapped back to have actual PHI text). 

294 :return: Precision value. 

295 """ 

296 # Find True Positive (TP) and precision 

297 tp = [i for i in all_pos if i in gt] 

298 try: 

299 precision = len(tp) / len(all_pos) 

300 except ZeroDivisionError: 

301 precision = 0 

302 

303 return precision
