27 """ 

28 Entry point for Presidio Analyzer. 

29 

30 Orchestrating the detection of PII entities and all related logic. 

31 

32 :param registry: instance of type RecognizerRegistry 

33 :param nlp_engine: instance of type NlpEngine 

34 (for example SpacyNlpEngine) 

35 :param app_tracer: instance of type AppTracer, used to trace the logic 

36 used during each request for interpretability reasons. 

37 :param log_decision_process: bool,

39 :param default_score_threshold: Minimum confidence value 

40 for detected entities to be returned 

41 :param supported_languages: List of possible languages this engine could be run on.

43 :param context_aware_enhancer: instance of type ContextAwareEnhancer for enhancing 

44 confidence score based on context words, (LemmaContextAwareEnhancer will be created 

45 by default if None passed)

148 def analyze( 

149 self, 

150 text: str, 

151 language: str, 

152 entities: Optional[List[str]] = None, 

153 correlation_id: Optional[str] = None, 

154 score_threshold: Optional[float] = None, 

155 return_decision_process: Optional[bool] = False, 

156 ad_hoc_recognizers: Optional[List[EntityRecognizer]] = None, 

157 context: Optional[List[str]] = None, 

158 allow_list: Optional[List[str]] = None, 

159 allow_list_match: Optional[str] = "exact", 

160 regex_flags: Optional[int] = re.DOTALL | re.MULTILINE | re.IGNORECASE, 

161 nlp_artifacts: Optional[NlpArtifacts] = None, 

162 ) -> List[RecognizerResult]: 

163 """ 

164 Find PII entities in text using different PII recognizers for a given language. 

165 

166 :param text: the text to analyze 

167 :param language: the language of the text 

168 :param entities: List of PII entities that should be looked for in the text. 

169 If entities=None then all entities are looked for. 

170 :param correlation_id: cross call ID for this request 

171 :param score_threshold: A minimum value for which 

172 to return an identified entity 

173 :param return_decision_process: Whether the analysis decision process steps 

174 returned in the response. 

175 :param ad_hoc_recognizers: List of recognizers which will be used only 

176 for this specific request. 

177 :param context: List of context words to enhance confidence score if matched 

178 with the recognized entity's recognizer context 

179 :param allow_list: List of words that the user defines as being allowed to keep 

180 in the text 

181 :param allow_list_match: How the allow_list should be interpreted; either as "exact" or as "regex". 

182 - If `regex`, results which match with any regex condition in the allow_list would be allowed and not be returned as potential PII. 

183 - if `exact`, results which exactly match any value in the allow_list would be allowed and not be returned as potential PII. 

184 :param regex_flags: regex flags to be used for when allow_list_match is "regex" 

185 :param nlp_artifacts: precomputed NlpArtifacts 

186 :return: an array of the found entities in the text 

187 

188 :Example: 

189 

190 ```python 

191 from presidio_analyzer import AnalyzerEngine 

192 

193 # Set up the engine, loads the NLP module (spaCy model by default) 

194 # and other PII recognizers 

195 analyzer = AnalyzerEngine() 

196 

197 # Call analyzer to get results 

198 results = analyzer.analyze(text='My phone number is 212-555-5555', entities=['PHONE_NUMBER'], language='en') 

199 print(results) 

200 ``` 

201 

202 """ # noqa: E501 

203 

204 all_fields = not entities 

205 

206 recognizers = self.registry.get_recognizers( 

207 language=language, 

208 entities=entities, 

209 all_fields=all_fields, 

210 ad_hoc_recognizers=ad_hoc_recognizers, 

211 ) 

212 

213 if all_fields: 

214 # Since all_fields=True, list all entities by iterating 

215 # over all recognizers 

216 entities = self.get_supported_entities(language=language) 

217 

218 # run the nlp pipeline over the given text, store the results in 

219 # a NlpArtifacts instance 

220 if not nlp_artifacts: 

221 nlp_artifacts = self.nlp_engine.process_text(text, language) 

222 

223 if self.log_decision_process: 

224 self.app_tracer.trace( 

225 correlation_id, "nlp artifacts:" + nlp_artifacts.to_json() 

226 ) 

227 

228 results = [] 

229 for recognizer in recognizers: 

230 # Lazy loading of the relevant recognizers 

231 if not recognizer.is_loaded: 

232 recognizer.load() 

233 recognizer.is_loaded = True 

234 

235 # analyze using the current recognizer and append the results 

236 current_results = recognizer.analyze( 

237 text=text, entities=entities, nlp_artifacts=nlp_artifacts 

238 ) 

239 if current_results: 

240 # add recognizer name to recognition metadata inside results 

241 # if not exists 

242 self.__add_recognizer_id_if_not_exists(current_results, recognizer) 

243 results.extend(current_results) 

244 

245 results = self._enhance_using_context( 

246 text, results, nlp_artifacts, recognizers, context 

247 ) 

248 

249 if self.log_decision_process: 

250 self.app_tracer.trace( 

251 correlation_id, 

252 json.dumps([str(result.to_dict()) for result in results]), 

253 ) 

254 

255 # Remove duplicates or low score results 

256 results = EntityRecognizer.remove_duplicates(results) 

257 results = self.__remove_low_scores(results, score_threshold) 

258 

259 if allow_list: 

260 results = self._remove_allow_list( 

261 results, allow_list, text, regex_flags, allow_list_match 

262 ) 

263 

264 if not return_decision_process: 

265 results = self.__remove_decision_process(results) 

266 

267 return results

269 def _enhance_using_context( 

270 self, 

271 text: str, 

272 raw_results: List[RecognizerResult], 

273 nlp_artifacts: NlpArtifacts, 

274 recognizers: List[EntityRecognizer], 

275 context: Optional[List[str]] = None, 

276 ) -> List[RecognizerResult]: 

277 """ 

278 Enhance confidence score using context words. 

279 

280 :param text: The actual text that was analyzed 

281 :param raw_results: Recognizer results which didn't take 

282 context into consideration 

283 :param nlp_artifacts: The nlp artifacts contains elements 

284 such as lemmatized tokens for better 

285 accuracy of the context enhancement process 

286 :param recognizers: the list of recognizers 

287 :param context: list of context words 

288 """ 

289 results = [] 

290 

291 for recognizer in recognizers: 

292 recognizer_results = [ 

293 r 

294 for r in raw_results 

295 if r.recognition_metadata[RecognizerResult.RECOGNIZER_IDENTIFIER_KEY] 

296 == recognizer.id 

297 ] 

298 other_recognizer_results = [ 

299 r 

300 for r in raw_results 

301 if r.recognition_metadata[RecognizerResult.RECOGNIZER_IDENTIFIER_KEY] 

302 != recognizer.id 

303 ] 

304 

305 # enhance score using context in recognizer level if implemented 

306 recognizer_results = recognizer.enhance_using_context( 

307 text=text, 

308 # each recognizer will get access to all recognizer results 

309 # to allow related entities contex enhancement 

310 raw_recognizer_results=recognizer_results, 

311 other_raw_recognizer_results=other_recognizer_results, 

312 nlp_artifacts=nlp_artifacts, 

313 context=context, 

314 ) 

315 

316 results.extend(recognizer_results) 

317 

318 # Update results in case surrounding words or external context are relevant to 

319 # the context words. 

320 results = self.context_aware_enhancer.enhance_using_context( 

321 text=text, 

322 raw_results=results, 

323 nlp_artifacts=nlp_artifacts, 

324 recognizers=recognizers, 

325 context=context, 

326 ) 

327 

328 return results

346 @staticmethod 

347 def _remove_allow_list( 

348 results: List[RecognizerResult], 

349 allow_list: List[str], 

350 text: str, 

351 regex_flags: Optional[int], 

352 allow_list_match: str, 

353 ) -> List[RecognizerResult]: 

354 """ 

355 Remove results which are part of the allow list. 

356 

357 :param results: List of RecognizerResult 

358 :param allow_list: list of allowed terms 

359 :param text: the text to analyze 

360 :param regex_flags: regex flags to be used for when allow_list_match is "regex" 

361 :param allow_list_match: How the allow_list 

362 should be interpreted; either as "exact" or as "regex" 

363 :return: List[RecognizerResult] 

364 """ 

365 new_results = [] 

366 if allow_list_match == "regex": 

367 pattern = "|".join(allow_list) 

368 re_compiled = re.compile(pattern, flags=regex_flags) 

369 

370 for result in results: 

371 word = text[result.start : result.end] 

372 

373 # if the word is not specified to be allowed, keep in the PII entities 

374 if not re_compiled.search(word): 

375 new_results.append(result) 

376 elif allow_list_match == "exact": 

377 for result in results: 

378 word = text[result.start : result.end] 

379 

380 # if the word is not specified to be allowed, keep in the PII entities 

381 if word not in allow_list: 

382 new_results.append(result) 

383 else: 

384 raise ValueError( 

385 "allow_list_match must either be set to 'exact' or 'regex'." 

386 ) 

387 

388 return new_results
