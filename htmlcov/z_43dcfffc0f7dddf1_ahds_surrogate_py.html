204 def operate(self, text: str = None, params: Dict = None) -> str: 

205 """ 

206 Replace PII entities using the AHDS de-identification service. 

207 

208 This operator uses AHDS de-identification service surrogation 

209 to generate contextually appropriate replacements for detected 

210 PII entities, maintaining text readability while protecting 

211 sensitive information. 

212 

213 :param text: The full text being processed 

214 :param params: Parameters including: 

215 - endpoint: AHDS de-identification service endpoint 

216 (optional, can use AHDS_ENDPOINT env var) 

217 - entities: List of entities detected by analyzer 

218 - input_locale: Input locale (default: "en-US") 

219 - surrogate_locale: Surrogate locale (default: "en-US") 

220 :return: Text with surrogates generated by AHDS 

221 """ 

222 

223 if not DeidentificationClient: 

224 raise ImportError( 

225 "Azure Health Data Services de-identification " 

226 "service SDK is not available. " 

227 "Please install azure-health-deidentification " 

228 "and azure-identity." 

229 ) 

230 

231 if not text: 

232 return "" 

233 

234 endpoint = params.get(self.ENDPOINT) or os.getenv("AHDS_ENDPOINT") 

235 if not endpoint: 

236 raise InvalidParamError( 

237 "AHDS de-identification service endpoint is required. " 

238 "Provide via 'endpoint' parameter " 

239 "or set AHDS_ENDPOINT environment variable." 

240 ) 

241 

242 entities = params.get(self.ENTITIES, []) 

243 input_locale = params.get(self.INPUT_LOCALE, "en-US") 

244 surrogate_locale = params.get(self.SURROGATE_LOCALE, "en-US") 

245 

246 # Convert analyzer results to AHDS tagged entities 

247 tagged_entities = self._convert_to_tagged_entities(entities) 

248 

249 try: 

250 # Use environment-aware credential (DefaultAzureCredential for dev, 

251 # ChainedTokenCredential for production) 

252 credential = get_azure_credential() 

253 client = DeidentificationClient(endpoint, credential, 

254 api_version="2025-07-15-preview") 

255 

256 # Create tagged entity collection 

257 tagged_entity_collection = TaggedPhiEntities( 

258 encoding=TextEncodingType.CODE_POINT, 

259 entities=tagged_entities 

260 ) 

261 

262 customizations = None 

263 if input_locale or surrogate_locale: 

264 customizations = DeidentificationCustomizationOptions() 

265 if input_locale: 

266 customizations.input_locale = input_locale 

267 if surrogate_locale: 

268 customizations.surrogate_locale = surrogate_locale 

269 

270 content = DeidentificationContent( 

271 input_text=text, 

272 operation_type=DeidentificationOperationType.SURROGATE_ONLY, 

273 tagged_entities=tagged_entity_collection, 

274 customizations=customizations 

275 ) 

276 

277 result = client.deidentify_text(content) 

278 if not result.output_text: 

279 raise InvalidParamError("Operation returned empty output text.") 

280 if result.output_text == text: 

281 raise InvalidParamError("Operation returned input text.") 

282 return result.output_text 

283 except Exception as e: 

284 raise InvalidParamError(f"AHDS Surrogate operation failed: {e}")

286 def _convert_to_tagged_entities(self, entities: List) -> List: 

287 """Convert analyzer results to AHDS SimplePhiEntity format.""" 

288 tagged_entities = [] 

289 

290 for entity in entities: 

291 # Handle both RecognizerResult objects and dict format 

292 if hasattr(entity, 'entity_type'): 

293 # RecognizerResult object 

294 category_name = entity.entity_type 

295 start = entity.start 

296 end = entity.end 

297 else: 

298 # Dict format 

299 category_name = entity.get('entity_type', entity.get('category', '')) 

300 start = entity.get('start', entity.get('offset', 0)) 

301 end = entity.get('end', start + entity.get('length', 0)) 

302 

303 # Map entity type to PhiCategory 

304 phi_category = self._map_to_phi_category(category_name) 

305 

306 tagged_entity = SimplePhiEntity( 

307 category=phi_category, 

308 offset=start, 

309 length=end - start 

310 ) 

311 tagged_entities.append(tagged_entity) 

312 

313 return tagged_entities

315 def _map_to_phi_category(self, entity_type: str): 

316 """Map Presidio entity types to AHDS PhiCategory. 

317 

318 Maps common Presidio PII entity types to their most appropriate AHDS PhiCategory 

319 counterparts for surrogate generation. 

320 

321 :param entity_type: The Presidio entity type (e.g., 'PERSON', 'PHONE_NUMBER') 

322 :return: Corresponding PhiCategory enum value 

323 """ 

324 # Convert to uppercase for case-insensitive matching 

325 entity_upper = entity_type.upper() 

326 

327 # Try exact match first 

328 if entity_upper in self.ENTITY_TYPE_MAPPING: 

329 category_name = self.ENTITY_TYPE_MAPPING[entity_upper] 

330 return self._get_safe_phi_category(category_name) 

331 

332 return PhiCategory.UNKNOWN 

333 

334 def _get_safe_phi_category(self, category_name: str): 

335 """Safely get PhiCategory enum value, with fallback to UNKNOWN.""" 

336 try: 

337 # Try to get the category by name 

338 return getattr(PhiCategory, category_name) 

339 except AttributeError: 

340 return PhiCategory.UNKNOWN

342 def validate(self, params: Dict = None) -> None: 

343 """Validate operator parameters.""" 

344 if not DeidentificationClient: 

345 raise ImportError( 

346 "Azure Health Data Services SDK is not available. " 

347 "Please install azure-health-deidentification and azure-identity." 

348 ) 

349 

350 if params is None: 

351 params = {} 

352 

353 endpoint = params.get(self.ENDPOINT) or os.getenv("AHDS_ENDPOINT") 

354 if not endpoint: 

355 raise InvalidParamError( 

356 "AHDS endpoint is required. Provide via 'endpoint' parameter " 

357 "or set AHDS_ENDPOINT environment variable." 

358 ) 

359 

360 entities = params.get(self.ENTITIES, []) 

361 if not isinstance(entities, list): 

362 raise InvalidParamError("Entities must be a list")
